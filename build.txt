name:
	Relic Package Manager
version:
	0.*
build:
    &DEV:OSSILE
    
    CRTSRCPF FILE(&INTOLIB/QSOURCE) RCDLEN(112)
            
    //Compile programs
    CRTSQLRPGI OBJ(&INTOLIB/BUILD) SRCSTMF('&DIR/QSOURCE/BUILD.SQLRPGLE') REPLACE(*YES) COMMIT(*NONE) TEXT('&NAME by &DEV')
    CRTSQLRPGI OBJ(&INTOLIB/RELIC) SRCSTMF('&DIR/QSOURCE/RELIC.SQLRPGLE') REPLACE(*YES) COMMIT(*NONE) TEXT('&NAME by &DEV')
    
    //Create commands
    CPYFRMSTMF FROMSTMF('&DIR/QSOURCE/RELICGET.CMD') TOMBR('/QSYS.lib/&INTOLIB.lib/QSOURCE.file/RELICGET.mbr') MBROPT(*ADD)
    CPYFRMSTMF FROMSTMF('&DIR/QSOURCE/RELICBLD.CMD') TOMBR('/QSYS.lib/&INTOLIB.lib/QSOURCE.file/RELICBLD.mbr') MBROPT(*ADD)
    CRTCMD CMD(&INTOLIB/RELICGET) PGM(&INTOLIB/RELIC) SRCFILE(&INTOLIB/QSOURCE) SRCMBR(RELICGET) REPLACE(*YES) TEXT('&NAME Command')
    CRTCMD CMD(&INTOLIB/RELICBLD) PGM(&INTOLIB/BUILD) SRCFILE(&INTOLIB/QSOURCE) SRCMBR(RELICBLD) REPLACE(*YES) TEXT('&NAME Build Command')
    CRTCMD CMD(&INTOLIB/RBLD) PGM(&INTOLIB/BUILD) SRCFILE(&INTOLIB/QSOURCE) SRCMBR(RELICBLD) REPLACE(*YES) TEXT('&NAME Build Command')
    
    DLTOBJ OBJ(&INTOLIB/QSOURCE) OBJTYPE(*FILE)
